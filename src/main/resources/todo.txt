SELECT id, name
FROM projects JOIN tags ON projects.id = tags.project_id
WHERE tag IN ('tag1', 'tag3')
GROUP BY projects.id
HAVING count(*) = 2;



SELECT
  projects.id, projects.name, projects.client,
  projects.status, projects.description, projects.githubRepo,
  projects.startedAt, projects.closedAt,
  tags.tag
FROM
  projects, tags
WHERE
  projects.id = tags.project_id
ORDER BY
  projects.id



CREATE TABLE tags
(
  project_id BIGINT,
  tag VARCHAR(63),

  PRIMARY KEY(project_id, tag)
);
